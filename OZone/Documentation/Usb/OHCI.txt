OHCIController.Frame()
{
	while(FrameRemaining > PeriodicStart)
	{
		ProcessControlList(Ratio);
		ProcessBulkList();
	}

	ProcessInterruptList(FrameCount & 0x1f);

	while(FrameRemaining > 0)
	{
		ProcessControlList(Ratio);
		ProcessBulkList();
	}
}
