Example 1:
1/1/1900 = 693595

1900 - 1 = 1899

1899 / 400 = 4 (+299)
4 * 146097 = 584388

299 / 100 = 2 (+99) (use standard table)
2 * 36524 = 73048

99 / 4 = 24 (+3)
24 * 1461 = 35064

3 * 365 = 1095

1 - 1 = 0

StandardStart[0] = 0

1 - 1 = 0

584388 + 73048 + 35064 + 1095 + 0 + 0 = 693595

Example 2:
2/29/2008 = 733100

2008 - 1 = 2007

2007 / 400 = 5 (+7)
5 * 146097 = 730485

7 / 100 = 0 (+7)
0 * 36524 = 0

7 / 4 = 1 (+3) (use leap year month table)
1 * 1461 = 1461

3 * 365 = 1095

2 - 1 = 1

LeapStart[1] = 31

29 - 1 = 28

730485 + 1461 + 1095 + 31 + 28 = 733100
